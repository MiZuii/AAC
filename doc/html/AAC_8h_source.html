60  return "[AAC] Invalid pixel type";
61  case error_codes::INVALID_PATH:

63  case error_codes::INVALID_ARGUMENTS:
64  return "[AAC] Invalid arguments";
65  case error_codes::IMAGE_OPEN_FAIL:
66  return "[AAC] Failed to open image";
67  case error_codes::IMAGE_ALLOCATION_ERROR:
68  return "[AAC] Failed to allocate memory for piexels array";
69  case error_codes::BRIGHTNESS_CALCULATION_FAIL:
70  return "[AAC] Failed to calculate image brightness array";
71  case error_codes::MATRIX_ALLOCATION_ERROR:
72  return "[AAC] Failed to allocate space for Matrix";

74  return "[AAC] Index out of range";
75  case error_codes::CHUNK_SIZE_ERROR:

77  default:
